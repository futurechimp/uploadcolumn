    def store_dir(relative = false)
      store_dir = @instance.send("#{@attr}_store_dir")
      store_dir ||= options[:store_dir]
      store_dir = File.join( store_dir, @instance.id.to_s ) if options[:store_dir_append_id]
      store_dir = File.join(options[:root_path], store_dir) unless relative
      store_dir
    end
