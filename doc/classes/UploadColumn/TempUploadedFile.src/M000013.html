    def upload(file)
      @tmp_dir = UploadColumn.generate_temp_name
      @tmp_full_dir = File.join(tmp_base_dir, @tmp_dir)      
      
      FileUtils.mkpath(@tmp_full_dir)
      
      @filename = UploadColumn::sanitize_filename(file.original_filename)

      # tries to identify and strip the extension of filename
      # if an regular expresion from EXT_REGEXPS matches and the
      # downcased extension is a known extension (in @options[:extensions])
      if @options[:fix_file_extensions]
        content_type = get_content_type(file.content_type.chomp) if file.content_type
        if content_type and @options[:mime_extensions][content_type]
          self.filename = self.filename_base + "." + @options[:mime_extensions][content_type]
        end 
      end

      local_file_path = File.join(@tmp_full_dir, self.filename)
      
      # stored uploaded file into local_file_path
      # If it was a Tempfile object, the temporary file will be
      # cleaned up automatically, so we do not have to care for this
      if file.respond_to?(:local_path) and file.local_path and File.exists?(file.local_path)
        FileUtils.copy_file(file.local_path, local_file_path)
      elsif file.respond_to?(:read)
        File.open(local_file_path, "wb") { |f| f.write(file.read) }
      else
        raise ArgumentError.new("Do not know how to handle #{file.inspect}")
      end


      # Multiple versions requested
      if options[:versions]
        # If it's an array, just copy without any processing
        if options[:versions].class == Array
          # Make a copy for each version
          options[:versions].each do |version|
            FileUtils.cp File.join(@tmp_full_dir, self.filename), File.join(@tmp_full_dir, self.filename_with_suffix(version.to_s))
          end
        else
          options[:versions].each_pair do |version, value|
            FileUtils.cp File.join(@tmp_full_dir, self.filename), File.join(@tmp_full_dir, self.filename_with_suffix(version.to_s))
            # If it's a hash and the key is a String like "200x150" or something, then process
            # the image to limit its size.
            if value.class == String #and value =~ /[d]{1,4}x[d]{1,4}/
              img = ::Magick::Image::read(absolute_path_with_suffix(version.to_s)).first
              img = img.change_geometry(value) do |c, r, i|
                i.resize(c,r)
              end
              img.write absolute_path_with_suffix(version.to_s)
            # If it's a symbol, call the method with the name of that symbol
            # passing the image as param, then save it.
            elsif value.class == Symbol and value != :none
              img = ::Magick::Image::read(absolute_path_with_suffix(version.to_s)).first
              img = @instance.send( value, img )
              img.write absolute_path_with_suffix(version.to_s)
            elsif value != :none
              raise TypeError
            end
          end          
        end
      end
      
      @instance[@attr] = filename
    end
